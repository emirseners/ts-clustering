"""
Plotting utilities for clustering results.

This module provides functions for visualizing clustering results including
dendrograms and cluster plots using Plotly for better performance.
"""
import matplotlib.pyplot as plt
import numpy as np
import math
from typing import List, TYPE_CHECKING
from scipy.cluster.hierarchy import dendrogram
import plotly.graph_objects as go
from plotly.subplots import make_subplots
import plotly.io as pio

if TYPE_CHECKING:
    from .clusterer import Cluster


def _plot_dendrogram(z: np.ndarray) -> None:
    """
    Plot hierarchical clustering dendrogram using matplotlib.
    
    This function visualizes the hierarchical clustering structure as a dendrogram,
    showing how clusters are formed at different distances.
    
    Parameters
    ----------
    z : np.ndarray
        Linkage matrix returned by scipy.cluster.hierarchy.linkage.
    """
    dendrogram(z, truncate_mode='lastp', show_leaf_counts=True, show_contracted=True)
    plt.show()

def plot_clusters(cluster_list: List["Cluster"], dist: str, mode: str = 'show', fname: str = 'results') -> None:
    """
    Plot cluster members on separate subplots using matplotlib.

    This function creates a grid of subplots where each subplot shows all time series
    belonging to a specific cluster. Each cluster is displayed in its own subplot
    with all member time series overlaid.

    Parameters
    ----------
    cluster_list : List[Cluster]
        List of Cluster objects containing clustered time series data.
    dist : str
        Distance metric name used for clustering. This will be displayed in the
        window title and can help identify which distance method was used.
    mode : str, default='show'
        Display mode for the plot:
        
        - 'show': Display the plot interactively using matplotlib.pyplot.show()
        - 'save': Save the plot to a PNG file without displaying it
        
    fname : str, default='results'
        Base filename for saving the plot (without extension). Only used when
        mode='save'. The file will be saved as '{fname}.png'.
    """
    main_fig = plt.figure(figsize=(14,10))
    main_fig.canvas.manager.set_window_title(dist + ' distance')
    no_plots = len(cluster_list)
    no_cols = 4
    no_rows = int(math.ceil(float(no_plots) / no_cols))
    i = 1
    
    for clust in cluster_list:
        sub_plot = main_fig.add_subplot(no_rows, no_cols, i)
        i = i + 1

        for j in clust.list_of_members:
            t = np.arange(j[1].shape[0])
            sub_plot.plot(t, j[1], linewidth=2)
        
        plt.title('Cluster no: ' + str(clust.cluster_id), weight='bold')

    plt.tight_layout()
    if mode=='show':
        plt.show()
    elif mode=='save':
        plt.savefig('{0}.png'.format(fname))

def interactive_plot_clusters(cluster_list: List["Cluster"], dist: str = "") -> go.Figure:
    """
    Create an interactive plot of cluster members using Plotly.
    
    This function creates an interactive visualization where each cluster is displayed
    in its own subplot. Users can interact with the plot through clicking, hovering,
    zooming, and panning. The plot includes an information panel that displays
    details when elements are clicked.
    
    Features
    --------
    - **Interactive Time Series**: Click on any time series line to see detailed information
    - **Cluster Information**: Double-click on cluster areas to see cluster metadata
    - **Hover Details**: Hover over lines to see data values and series information  
    - **Zoom and Pan**: Interactive navigation with zoom and pan controls
    - **Performance**: Optimized for large datasets using Plotly's efficient rendering
    - **Information Panel**: Dynamic panel showing clicked element details
    
    Parameters
    ----------
    cluster_list : List[Cluster]
        List of Cluster objects containing the clustered time series data.
        Each cluster should have members with (label, data_array) tuples.
    dist : str, optional
        Distance metric name used for clustering. This will be displayed in the
        plot title to help identify which distance method was used. Default is "".
        
    Returns
    -------
    plotly.graph_objects.Figure
        Interactive Plotly figure object that can be displayed, saved, or further customized.
    """
    no_plots = len(cluster_list)
    no_cols = 3
    no_rows = int(math.ceil(float(no_plots) / no_cols))
    
    # Create subplots with Plotly
    # Build specs array for subplots
    specs = []
    for _ in range(no_rows):
        specs.append([{"secondary_y": False}] * no_cols)
    
    # Add info panel row that spans all columns
    info_row = [{"colspan": no_cols, "secondary_y": False}]
    info_row.extend([None] * (no_cols - 1))
    specs.append(info_row)
    
    fig = make_subplots(
        rows=no_rows + 1,  # +1 for info panel
        cols=no_cols,
        subplot_titles=[f'Cluster {clust.cluster_id} ({clust.number_of_members} members)' for clust in cluster_list] + ['Information Panel'],
        specs=specs,
        vertical_spacing=0.08,
        horizontal_spacing=0.05
    )
    
    # Create cluster subplots
    for i, clust in enumerate(cluster_list):
        row_idx = i // no_cols + 1
        col_idx = i % no_cols + 1
        
        # Plot time series for this cluster
        for j, member in enumerate(clust.list_of_members):
            label, data = member
            t = np.arange(data.shape[0])
            
            # Create detailed hover text with statistics
            data_stats = {
                'min': float(np.min(data)),
                'max': float(np.max(data)),
                'mean': float(np.mean(data)),
                'std': float(np.std(data))
            }
            
            hover_text = (f"<b>{label}</b><br>" +
                         f"Cluster: {clust.cluster_id}<br>" +
                         f"Length: {len(data)} points<br>" +
                         f"Range: [{data_stats['min']:.3f}, {data_stats['max']:.3f}]<br>" +
                         f"Mean: {data_stats['mean']:.3f}<br>" +
                         f"Std: {data_stats['std']:.3f}<br>" +
                         f"<i>Click for more details</i>")
            
            # Create customdata with all necessary information
            customdata_item = {
                'label': label,
                'data': data.tolist(),
                'cluster_id': clust.cluster_id,
                'cluster_size': clust.number_of_members,
                'stats': data_stats,
                'member_index': j
            }
            
            fig.add_trace(
                go.Scatter(
                    x=t,
                    y=data,
                    mode='lines',
                    name=f'{label}',
                    line=dict(width=2.5, color=f'rgba({(i*50) % 255}, {(j*80) % 255}, {(i+j)*60 % 255}, 0.8)'),
                    hoverinfo='text',
                    hovertext=hover_text,
                    customdata=[customdata_item] * len(t),
                    hovertemplate='%{hovertext}<extra></extra>',
                    showlegend=False
                ),
                row=row_idx, col=col_idx
            )
    
    # Create information panel with initial instructions
    initial_info = (
        "<b>Interactive Plot Instructions:</b><br><br>" +
        "🔵 <b>Single Click</b> on any line → View time series details<br>" +
        "🔵 <b>Double Click</b> on any line → View cluster information<br>" +
        "🔵 <b>Hover</b> over lines → Quick statistics preview<br>" +
        "🔵 <b>Zoom & Pan</b> → Use mouse wheel and drag<br><br>" +
        "<i>Click on any time series line to see detailed information here!</i>"
    )
    
    # Add information panel as a text box in the bottom subplot
    fig.add_trace(
        go.Scatter(
            x=[0.5],
            y=[0.5],
            mode='text',
            text=[initial_info],
            textposition='middle center',
            textfont=dict(size=14, color='#2c3e50'),
            showlegend=False,
            hoverinfo='skip'
        ),
        row=no_rows + 1, col=1
    )
    
    # Update the information panel subplot to remove axes and background
    fig.update_xaxes(showgrid=False, showticklabels=False, zeroline=False, row=no_rows + 1, col=1)
    fig.update_yaxes(showgrid=False, showticklabels=False, zeroline=False, row=no_rows + 1, col=1)
    
    # Update layout with better styling
    fig.update_layout(
        title=dict(
            text=f'<b>{dist} Distance - Interactive Clustering Plot</b>',
            x=0.5,
            font=dict(size=18, color='#2c3e50')
        ),
        height=900,
        showlegend=False,
        hovermode='closest',
        template='plotly_white',
        paper_bgcolor='#f8f9fa',
        plot_bgcolor='white'
    )
    
    # Update axes for cluster subplots
    for i in range(1, no_rows + 1):
        for j in range(1, no_cols + 1):
            if (i-1) * no_cols + j <= no_plots:
                fig.update_xaxes(
                    title_text="Time",
                    showgrid=True,
                    gridcolor='rgba(128,128,128,0.2)',
                    row=i, col=j
                )
                fig.update_yaxes(
                    title_text="Value",
                    showgrid=True,
                    gridcolor='rgba(128,128,128,0.2)',
                    row=i, col=j
                )
    
    # Configure the plot for better interactivity
    fig.update_layout(
        clickmode='event+select',
        dragmode='pan',
        modebar=dict(
            remove=['lasso2d', 'select2d'],
            add=['pan', 'zoom', 'reset', 'autoScale2d']
        )
    )
    
    # Instead of trying to inject JavaScript, we'll use Plotly's post-processing approach
    # Save the configuration for JavaScript injection during HTML export
    fig._custom_js = _generate_interaction_js()
    
    # Return the figure object for further customization or display
    return fig




def _generate_interaction_js() -> str:
    """Generate JavaScript code for interactive functionality."""
    return """
    <script>
    document.addEventListener('DOMContentLoaded', function() {
        // Wait for Plotly to load
        var checkPlotly = setInterval(function() {
            if (typeof Plotly !== 'undefined') {
                clearInterval(checkPlotly);
                initializeInteractivity();
            }
        }, 100);
        
        function initializeInteractivity() {
            var plotElement = document.querySelector('.plotly-graph-div');
            if (!plotElement) return;
            
            var gd = plotElement;
            
            // Single click handler for time series details
            gd.on('plotly_click', function(data) {
                if (data.points && data.points.length > 0) {
                    var point = data.points[0];
                    var customData = point.customdata;
                    
                    if (customData) {
                        showTimeSeriesInfo(customData, point);
                    }
                }
            });
            
            // Double click handler for cluster information
            gd.on('plotly_doubleclick', function(data) {
                if (data.points && data.points.length > 0) {
                    var point = data.points[0];
                    var customData = point.customdata;
                    
                    if (customData) {
                        showClusterInfo(customData);
                    }
                }
            });
        }
        
        function showTimeSeriesInfo(customData, point) {
            var label = customData.label;
            var stats = customData.stats;
            var clusterId = customData.cluster_id;
            var dataArray = customData.data;
            
            var infoHtml = 
                '<b>📈 TIME SERIES DETAILS</b><br>' +
                '━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━<br>' +
                '<b>Label:</b> ' + label + '<br>' +
                '<b>Cluster:</b> ' + clusterId + '<br>' +
                '<b>Data Points:</b> ' + dataArray.length + '<br>' +
                '<b>Value Range:</b> [' + stats.min.toFixed(3) + ', ' + stats.max.toFixed(3) + ']<br>' +
                '<b>Mean:</b> ' + stats.mean.toFixed(3) + '<br>' +
                '<b>Std Dev:</b> ' + stats.std.toFixed(3) + '<br>' +
                '<b>First 5:</b> [' + dataArray.slice(0, 5).map(x => x.toFixed(2)).join(', ') + ']<br>' +
                '<b>Last 5:</b> [' + dataArray.slice(-5).map(x => x.toFixed(2)).join(', ') + ']<br><br>' +
                '<i>Double-click for cluster information</i>';
            
            updateInfoPanel(infoHtml);
        }
        
        function showClusterInfo(customData) {
            var clusterId = customData.cluster_id;
            var clusterSize = customData.cluster_size;
            
            var infoHtml = 
                '<b>🔵 CLUSTER INFORMATION</b><br>' +
                '━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━<br>' +
                '<b>Cluster ID:</b> ' + clusterId + '<br>' +
                '<b>Total Members:</b> ' + clusterSize + ' time series<br>' +
                '<b>Cluster Type:</b> Hierarchical clustering result<br><br>' +
                '📊 <b>Interaction Options:</b><br>' +
                '• Single click any line → View series details<br>' +
                '• Double click any line → View this cluster info<br>' +
                '• Hover over lines → Quick statistics<br>' +
                '• Use zoom and pan for detailed exploration<br><br>' +
                '<i>Single-click on any line for detailed time series information</i>';
            
            updateInfoPanel(infoHtml);
        }
        
        function updateInfoPanel(htmlContent) {
            // Find the information panel trace and update it
            var plotElement = document.querySelector('.plotly-graph-div');
            if (plotElement && plotElement.data) {
                // Look for the info panel trace (last trace)
                var traces = plotElement.data;
                var infoTraceIndex = traces.length - 1;
                
                // Update the text content
                Plotly.restyle(plotElement, {
                    'text': [[htmlContent]]
                }, [infoTraceIndex]);
            }
        }
    });
    </script>
    """


def export_interactive_plot(fig: go.Figure, filename: str, auto_open: bool = True) -> None:
    """
    Export an interactive clustering plot to HTML with proper JavaScript injection.
    
    This function exports the Plotly figure to an HTML file and properly injects
    the JavaScript code needed for the interactive functionality to work correctly.
    
    Parameters
    ----------
    fig : plotly.graph_objects.Figure
        The Plotly figure object created by interactive_plot_clusters().
    filename : str
        The filename for the HTML output (should end with .html).
    auto_open : bool, default=True
        Whether to automatically open the HTML file in the default browser.
    """
    # Generate the basic HTML
    html_string = pio.to_html(fig, include_plotlyjs='cdn')
    
    # Check if the figure has custom JavaScript
    if hasattr(fig, '_custom_js'):
        # Find the closing body tag and inject the JavaScript before it
        js_code = fig._custom_js
        
        # Insert the JavaScript just before the closing body tag
        if '</body>' in html_string:
            html_string = html_string.replace('</body>', js_code + '\n</body>')
        else:
            # Fallback: append at the end
            html_string += js_code
    
    # Write the HTML file
    with open(filename, 'w', encoding='utf-8') as f:
        f.write(html_string)
    
    # Auto-open if requested
    if auto_open:
        import webbrowser
        import os
        webbrowser.open('file://' + os.path.abspath(filename))
