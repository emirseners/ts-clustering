"""
Plotting utilities for clustering results.

This module provides functions for visualizing clustering results including
dendrograms and cluster plots using Plotly for better performance.
"""
import matplotlib.pyplot as plt
import numpy as np
import math
from typing import List, TYPE_CHECKING
from scipy.cluster.hierarchy import dendrogram
import plotly.graph_objects as go
from plotly.subplots import make_subplots

if TYPE_CHECKING:
    from .clusterer import Cluster


def _plot_dendrogram(z: np.ndarray) -> None:
    """
    Plot hierarchical clustering dendrogram using matplotlib.
    
    This function visualizes the hierarchical clustering structure as a dendrogram,
    showing how clusters are formed at different distances.
    
    Parameters
    ----------
    z : np.ndarray
        Linkage matrix returned by scipy.cluster.hierarchy.linkage.
    """
    dendrogram(z, truncate_mode='lastp', show_leaf_counts=True, show_contracted=True)
    plt.show()

def plot_clusters(cluster_list: List["Cluster"], dist: str, mode: str = 'show', fname: str = 'results') -> None:
    """
    Plot cluster members on separate subplots using matplotlib.

    This function creates a grid of subplots where each subplot shows all time series
    belonging to a specific cluster. Each cluster is displayed in its own subplot
    with all member time series overlaid.

    Parameters
    ----------
    cluster_list : List[Cluster]
        List of Cluster objects containing clustered time series data.
    dist : str
        Distance metric name used for clustering. This will be displayed in the
        window title and can help identify which distance method was used.
    mode : str, default='show'
        Display mode for the plot:
        
        - 'show': Display the plot interactively using matplotlib.pyplot.show()
        - 'save': Save the plot to a PNG file without displaying it
        
    fname : str, default='results'
        Base filename for saving the plot (without extension). Only used when
        mode='save'. The file will be saved as '{fname}.png'.
    """
    main_fig = plt.figure(figsize=(14,10))
    main_fig.canvas.manager.set_window_title(dist + ' distance')
    no_plots = len(cluster_list)
    no_cols = 4
    no_rows = int(math.ceil(float(no_plots) / no_cols))
    i = 1
    
    for clust in cluster_list:
        sub_plot = main_fig.add_subplot(no_rows, no_cols, i)
        i = i + 1

        for j in clust.members:
            t = np.arange(j[1].shape[0])
            sub_plot.plot(t, j[1], linewidth=2)
        
        plt.title('Cluster no: ' + str(clust.no), weight='bold')

    plt.tight_layout()
    if mode=='show':
        plt.show()
    elif mode=='save':
        plt.savefig('{0}.png'.format(fname))

def interactive_plot_clusters(cluster_list: List["Cluster"], dist: str = "") -> go.Figure:
    """
    Create an interactive plot of cluster members using Plotly.
    
    This function creates an interactive visualization where each cluster is displayed
    in its own subplot. Users can interact with the plot through clicking, hovering,
    zooming, and panning. The plot includes an information panel that displays
    details when elements are clicked.
    
    Features
    --------
    - **Interactive Time Series**: Click on any time series line to see detailed information
    - **Cluster Information**: Double-click on cluster areas to see cluster metadata
    - **Hover Details**: Hover over lines to see data values and series information  
    - **Zoom and Pan**: Interactive navigation with zoom and pan controls
    - **Performance**: Optimized for large datasets using Plotly's efficient rendering
    - **Information Panel**: Dynamic panel showing clicked element details
    
    Parameters
    ----------
    cluster_list : List[Cluster]
        List of Cluster objects containing the clustered time series data.
        Each cluster should have members with (label, data_array) tuples.
    dist : str, optional
        Distance metric name used for clustering. This will be displayed in the
        plot title to help identify which distance method was used. Default is "".
        
    Returns
    -------
    plotly.graph_objects.Figure
        Interactive Plotly figure object that can be displayed, saved, or further customized.
    """
    no_plots = len(cluster_list)
    no_cols = 3
    no_rows = int(math.ceil(float(no_plots) / no_cols))
    
    # Create subplots with Plotly
    # Build specs array for subplots
    specs = []
    for _ in range(no_rows):
        specs.append([{"secondary_y": False}] * no_cols)
    
    # Add info panel row that spans all columns
    info_row = [{"colspan": no_cols, "secondary_y": False}]
    info_row.extend([None] * (no_cols - 1))
    specs.append(info_row)
    
    fig = make_subplots(
        rows=no_rows + 1,  # +1 for info panel
        cols=no_cols,
        subplot_titles=[f'Cluster {clust.no} ({clust.size} members)' for clust in cluster_list] + ['Information Panel'],
        specs=specs,
        vertical_spacing=0.08,
        horizontal_spacing=0.05
    )
    
    # Store cluster data for click handling
    cluster_data = {}
    
    # Create cluster subplots
    for i, clust in enumerate(cluster_list):
        row_idx = i // no_cols + 1
        col_idx = i % no_cols + 1
        
        # Plot time series for this cluster
        for j, member in enumerate(clust.members):
            label, data = member
            t = np.arange(data.shape[0])
            
            # Create hover text with detailed information
            hover_text = f"Label: {label}<br>Cluster: {clust.no}<br>Data Length: {len(data)}<br>Hover over line for values"
            
            fig.add_trace(
                go.Scatter(
                    x=t,
                    y=data,
                    mode='lines',
                    name=f'{label} (Cluster {clust.no})',
                    line=dict(width=2),
                    hoverinfo='text',
                    hovertext=hover_text,
                    customdata=[(label, data.tolist(), clust.no, clust.size) for _ in range(len(t))],
                    hovertemplate='%{hovertext}<extra></extra>',
                    showlegend=False
                ),
                row=row_idx, col=col_idx
            )
        
        # Store cluster information
        cluster_data[f'cluster_{clust.no}'] = {
            'cluster': clust,
            'row': row_idx,
            'col': col_idx
        }
    
    # Create information panel
    info_text = ("Interactive Plot Instructions:<br><br>" +
                "[TS] Time Series: Hover over lines for details<br>" +
                "[CL] Cluster: Click on cluster areas for information<br><br>" +
                "Information will appear here when you interact.")
    
    fig.add_annotation(
        text=info_text,
        xref="paper", yref="paper",
        x=0.5, y=0.02,
        xanchor='center', yanchor='bottom',
        showarrow=False,
        font=dict(size=12, color='black'),
        bgcolor='lightblue',
        bordercolor='black',
        borderwidth=1,
        align='left'
    )
    
    # Update layout
    fig.update_layout(
        title=f'{dist} Distance - Interactive Plot',
        height=800,
        showlegend=False,
        hovermode='closest',
        template='plotly_white'
    )
    
    # Update axes
    for i in range(1, no_rows + 1):
        for j in range(1, no_cols + 1):
            fig.update_xaxes(title_text="Time", row=i, col=j)
            fig.update_yaxes(title_text="Value", row=i, col=j)
    
    # Add JavaScript-based click events that work in HTML files
    fig.update_traces(
        hoverinfo='text',
        hovertemplate='%{hovertext}<extra></extra>'
    )
    
    # Add click events using Plotly's built-in functionality
    # This approach is more reliable than JavaScript injection
    
    # Create a custom click event handler
    def handle_click(trace, points, state):
        if points.point_inds:
            point_idx = points.point_inds[0]
            custom_data = trace.customdata[point_idx]
            label, data_list, cluster_no, cluster_size = custom_data
            
            # Convert data back to numpy array for calculations
            data = np.array(data_list)
            
            # Display time series information
            info_str = f"[TS] TIME SERIES CLICKED:<br>"
            info_str += f"━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━<br>"
            info_str += f"Label: {label}<br>"
            info_str += f"Cluster: {cluster_no}<br>"
            info_str += f"Data Length: {len(data)}<br>"
            info_str += f"Data Range: [{data.min():.3f}, {data.max():.3f}]<br>"
            info_str += f"Mean: {data.mean():.3f}, Std: {data.std():.3f}<br>"
            info_str += f"First 4 data points: {data[:4].tolist()}<br>"
            info_str += f"Last 4 data points: {data[-4:].tolist()}"
            
            # Update information panel
            fig.update_annotations(
                dict(
                    text=info_str,
                    font=dict(size=11)
                ),
                selector=dict(text=info_text)
            )
    
    # Add click events to all traces
    for trace in fig.data:
        trace.on_click(handle_click)
    
    # Add a custom layout update to make the plot more interactive
    fig.update_layout(
        clickmode='event',
        dragmode='pan',  # Allow panning
        modebar=dict(
            remove=['lasso2d', 'select2d'],  # Remove selection tools that might interfere
            add=['pan', 'zoom', 'reset']
        )
    )
    
    # Add custom JavaScript for click events that will work in HTML files
    # We'll use a different approach - adding the JavaScript directly to the HTML output
    custom_js = """
    <script>
    // Wait for the plot to load
    window.addEventListener('load', function() {
        // Get the plot element
        var plotDiv = document.querySelector('.plotly-graph-div');
        if (plotDiv) {
            // Add click event listener to the plot
            plotDiv.on('plotly_click', function(data) {
                var point = data.points[0];
                var customData = point.customdata;
                
                if (customData && customData.length >= 4) {
                    var label = customData[0];
                    var clusterNo = customData[2];
                    var clusterSize = customData[3];
                    
                    // Create info string
                    var infoStr = '[TS] TIME SERIES CLICKED:<br>' +
                                 '━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━<br>' +
                                 'Label: ' + label + '<br>' +
                                 'Cluster: ' + clusterNo + '<br>' +
                                 'Data Length: ' + point.y.length + '<br>' +
                                 'Data Range: [' + Math.min(...point.y).toFixed(3) + ', ' + Math.max(...point.y).toFixed(3) + ']<br>' +
                                 'Mean: ' + (point.y.reduce((a, b) => a + b, 0) / point.y.length).toFixed(3) + '<br>' +
                                 'First 4 data points: [' + point.y.slice(0, 4).join(', ') + ']<br>' +
                                 'Last 4 data points: [' + point.y.slice(-4).join(', ') + ']';
                    
                    // Update the information panel
                    var infoPanel = document.querySelector('[data-unformatted*="Information will appear here when you interact"]');
                    if (infoPanel) {
                        infoPanel.innerHTML = infoStr;
                    }
                }
            });
            
            // Add double-click event for cluster information
            plotDiv.on('plotly_doubleclick', function(data) {
                var point = data.points[0];
                var customData = point.customdata;
                
                if (customData && customData.length >= 4) {
                    var clusterNo = customData[2];
                    var clusterSize = customData[3];
                    
                    var infoStr = '[CL] CLUSTER CLICKED:<br>' +
                                 '━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━<br>' +
                                 'Cluster No: ' + clusterNo + '<br>' +
                                 'Size: ' + clusterSize + '<br>' +
                                 'Click on time series lines for detailed information';
                    
                    // Update the information panel
                    var infoPanel = document.querySelector('[data-unformatted*="Information will appear here when you interact"]');
                    if (infoPanel) {
                        infoPanel.innerHTML = infoStr;
                    }
                }
            });
        }
    });
    </script>
    """
    
    # Add the custom JavaScript to the figure using a valid annotation
    fig.add_annotation(
        text=custom_js,
        xref="paper", yref="paper",
        x=0, y=0,
        showarrow=False,
        font=dict(size=1, color='rgba(0,0,0,0)'),
        bgcolor='rgba(0,0,0,0)',
        bordercolor='rgba(0,0,0,0)'
    )
    
    # Return the figure object for further customization or display
    return fig
