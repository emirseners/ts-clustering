def _plot_dendrogram(z: np.ndarray) -> None:
    """
    Plot hierarchical clustering dendrogram.
    
    Parameters
    ----------
        z : np.ndarray
            Linkage matrix
    """

    dendrogram(z, truncate_mode='lastp', show_leaf_counts=True, show_contracted=True)
    plt.show()

def plot_clusters(cluster_list: List["Cluster"], dist: str, mode: str = 'show', fname: str = 'results') -> None:
    """
    Plot cluster members on separate subplots.

    Parameters
    ----------
        cluster_list : List[Cluster]
            List of Cluster objects
        dist : str
                Distance metric name for title
        mode : str
            'show' or 'save'
        fname : str
            Filename for saving (without extension)
    """
    main_fig = plt.figure(figsize=(14,10))
    main_fig.canvas.manager.set_window_title(dist + ' distance')
    no_plots = len(cluster_list)
    no_cols = 4
    no_rows = int(math.ceil(float(no_plots) / no_cols))
    i = 1
    
    for clust in cluster_list:
        sub_plot = main_fig.add_subplot(no_rows, no_cols, i)
        i = i + 1

        for j in clust.members:
            t = np.arange(j[1].shape[0])
            sub_plot.plot(t, j[1], linewidth=2)
        
        plt.title('Cluster no: ' + str(clust.no), weight='bold')

    plt.tight_layout()
    if mode=='show':
        plt.show()
    elif mode=='save':
        plt.savefig('{0}.png'.format(fname))

def interactive_plot_clusters(cluster_list, dist=""):
    """
    Plot cluster members on separate subplots with interactive click callbacks.
    
    Parameters
    ----------
    cluster_list : List[Cluster]
        List of Cluster objects. Each Cluster should have attributes:
        - no : cluster ID
        - size : number of members
        - indices : list/array of member indices
        - sample : tuple (label, 1D numpy array) of the representative series
        - members : list of (label, 1D numpy array) tuples
    dist : str, optional
        Distance metric name for the overall title.
    """
    n_clusters = len(cluster_list)
    # Arrange subplots in rows of up to 3 columns
    cols = 3 if n_clusters > 1 else 1
    rows = (n_clusters + cols - 1) // cols

    # Create figure with shared x-axis (time) but separate y-axes
    fig = go.FigureWidget(make_subplots(
        rows=rows, cols=cols,
        subplot_titles=[f"Cluster {cl.no} (n={cl.size})" for cl in cluster_list],
        horizontal_spacing=0.08, vertical_spacing=0.15
    ))

    # Info panel (HTML) will show details on click
    info_panel = HTML(
        "<b>Interactive Plot Instructions:</b><br>"
        "Click on a time series line to see its label and stats.<br>"
        "Click on the background of a subplot to see cluster details."
    )

    # Add traces and metadata
    for idx, cluster in enumerate(cluster_list):
        row = idx // cols + 1
        col = idx % cols + 1

        # Plot each member series
        for label, data in cluster.members:
            x_vals = list(range(len(data)))
            fig.add_trace(
                go.Scattergl(
                    x=x_vals,
                    y=data,
                    mode="lines",
                    line=dict(width=1),
                    hovertemplate=f"<b>{label}</b><br>index: %{{x}}<br>value: %{{y}}<extra></extra>",
                    meta={
                        "type": "series",
                        "cluster_no": cluster.no,
                        "label": label,
                        "data": data.tolist(),
                        "cluster_meta": {
                            "size": cluster.size,
                            "indices": list(cluster.indices),
                            "members": [m[0] for m in cluster.members],
                            "sample_label": cluster.sample[0],
                            "sample_data": cluster.sample[1].tolist()
                        }
                    }
                ),
                row=row, col=col
            )

        # Plot the cluster representative series with thicker line
        rep_label, rep_data = cluster.sample
        fig.add_trace(
            go.Scattergl(
                x=list(range(len(rep_data))),
                y=rep_data,
                mode="lines",
                line=dict(width=3),
                hovertemplate=f"<b>Cluster {cluster.no} representative</b><br>index: %{{x}}<br>value: %{{y}}<extra></extra>",
                meta={
                    "type": "series",
                    "cluster_no": cluster.no,
                    "label": f"{rep_label} (rep)",
                    "data": rep_data.tolist(),
                    "cluster_meta": {
                        "size": cluster.size,
                        "indices": list(cluster.indices),
                        "members": [m[0] for m in cluster.members],
                        "sample_label": rep_label,
                        "sample_data": rep_data.tolist()
                    }
                }
            ),
            row=row, col=col
        )

        # Add an invisible scatter that covers the subplot area for cluster clicks
        # This gives us a clickable background to show cluster-level info
        fig.add_trace(
            go.Scattergl(
                x=[0, len(rep_data)],
                y=[min(rep_data), max(rep_data)],
                mode="markers",
                marker=dict(opacity=0),
                showlegend=False,
                hoverinfo="skip",
                meta={
                    "type": "cluster",
                    "cluster_no": cluster.no,
                    "cluster_meta": {
                        "size": cluster.size,
                        "indices": list(cluster.indices),
                        "members": [m[0] for m in cluster.members],
                        "sample_label": rep_label,
                        "sample_data": rep_data.tolist()
                    }
                }
            ),
            row=row, col=col
        )

    # Callback function for clicks
    def handle_click(trace, points, selector):
        if not points.point_inds:
            return

        m = trace.meta
        if m["type"] == "series":
            # Show time series-level info
            info_panel.value = (
                f"<b>Time Series Clicked</b><br>"
                f"<pre>"
                f"Cluster: {m['cluster_no']}\n"
                f"Label: {m['label']}\n"
                f"Length: {len(m['data'])}\n"
                f"Range: [{min(m['data']):.3f}, {max(m['data']):.3f}]\n"
                f"Mean: {sum(m['data'])/len(m['data']):.3f}\n"
                f"First 4 pts: {m['data'][:4]}\n"
                f"Last 4 pts: {m['data'][-4:]}\n"
                f"</pre>"
            )
        elif m["type"] == "cluster":
            # Show cluster-level info
            cmeta = m["cluster_meta"]
            info_panel.value = (
                f"<b>Cluster {m['cluster_no']} Clicked</b><br>"
                f"<pre>"
                f"Size: {cmeta['size']}\n"
                f"Indices: {cmeta['indices']}\n"
                f"Members: {cmeta['members']}\n"
                f"Representative label: {cmeta['sample_label']}\n"
                f"Representative data length: {len(cmeta['sample_data'])}\n"
                f"Rep range: [{min(cmeta['sample_data']):.3f}, {max(cmeta['sample_data']):.3f}]\n"
                f"Rep first 4 pts: {cmeta['sample_data'][:4]}\n"
                f"Rep last 4 pts: {cmeta['sample_data'][-4:]}\n"
                f"</pre>"
            )

    # Attach click callbacks to every trace
    for tr in fig.data:
        tr.on_click(handle_click)

    # Styling tweaks
    fig.update_layout(
        height=300 * rows + 120,
        width=1000,
        title=f"{dist} distance â€“ Interactive Clustering Plot",
        showlegend=False
    )
    for row in range(rows):
        for col in range(cols):
            fig.update_xaxes(title_text="Time", row=row+1, col=col+1)
            fig.update_yaxes(title_text="Value", row=row+1, col=col+1)

    # Display figure and info panel together
    # For standalone Python scripts, always save as HTML
    import plotly.io as pio
    import os
    
    # Save the plot as an HTML file
    html_file = "interactive_clustering_plot.html"
    fig.write_html(html_file)
    
    print(f"Plot saved as '{html_file}' in the current directory.")
    print("Please open this file in your web browser to view the interactive plot.")
    
    # Try to open in default browser
    try:
        import webbrowser
        webbrowser.open(f"file://{os.path.abspath(html_file)}")
        print("Attempting to open in default browser...")
    except Exception as e:
        print(f"Could not open browser automatically: {e}")
        print(f"Please manually open '{html_file}' in your web browser.")
    
    # Try to display in Jupyter environment if available
    try:
        from IPython.display import display
        display(VBox([fig, info_panel]))
    except ImportError:
        pass
    
    return VBox([fig, info_panel])
